import numpy as np

# create a one-dimensional array
numbers = np.array([1, 2, 3, 4, 5, 6])

# reshape(2,3) changes the array
# into 2 rows and 3 columns.

reshaped-array = numbers.reshape(2, 3)

# print original array
print("original array:")
print(numbers)

# print reshaped array
print("reshaped array:")
print(reshaped-array)
